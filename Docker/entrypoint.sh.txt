#!/bin/bash

# Start the Flask API (Backend) on an internal port (5000) in the background.
# The Streamlit app will call this API on http://localhost:5000.
echo "Starting RAG API (Gunicorn) on port 5000..."
gunicorn -w 4 -b 0.0.0.0:5000 app:app &

# Start the Streamlit Frontend on the container's exposed port (Port 80).
# This is the GUI the user sees via the ALB.
echo "Starting Streamlit Frontend on port 80..."
streamlit run frontend.py --server.port 80 --server.address 0.0.0.0

# Wait for background jobs to finish (keeps the container alive)
wait
