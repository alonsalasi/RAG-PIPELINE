import os
import boto3
import time
from flask import Flask, jsonify, request
from requests_aws4auth import AWS4Auth
from opensearchpy import OpenSearch, RequestsHttpConnection

# --- RAG/Vector Imports ---
from langchain.embeddings import SentenceTransformerEmbeddings
from langchain_community.vectorstores import OpenSearchVectorSearch
from langchain.chains import RetrievalQA
from langchain_community.llms import Bedrock
from langchain.schema.document import Document

app = Flask(__name__)

# --- Configuration (from ECS Task Definition) ---
# This is the address of the OpenSearch domain (e.g., os-domain-abc.us-west-2.es.amazonaws.com)
VECTOR_DB_ENDPOINT = os.environ.get("VECTOR_DB_ENDPOINT") 
AWS_REGION = os.environ.get("AWS_REGION", "us-west-2")
EMBEDDING_MODEL_NAME = 'all-MiniLM-L6-v2' 
OPENSEARCH_INDEX_NAME = 'rag-vector-index'
BEDROCK_MODEL_ID = 'anthropic.claude-v2' 

# --- Global Client Initialization ---
# Initialize RAG components once
embeddings_model = SentenceTransformerEmbeddings(model_name=EMBEDDING_MODEL_NAME)
# Initialize the Bedrock client to invoke the LLM
bedrock_client = boto3.client('bedrock-runtime', region_name=AWS_REGION)

def get_aws_auth_header():
    """Generates AWS V4 Signature for OpenSearch access using ECS Task Role credentials."""
    session = boto3.Session()
    credentials = session.get_credentials()
    # Remove 'https://' for hostname signing
    host = VECTOR_DB_ENDPOINT.replace('https://', '') 

    return AWS4Auth(
        credentials.access_key,
        credentials.secret_key,
        AWS_REGION,
        'es', # Service name for OpenSearch is 'es'
        session_token=credentials.token
    )

def initialize_retriever():
    """Initializes the OpenSearch client and LangChain Retriever."""
    if not VECTOR_DB_ENDPOINT:
        raise EnvironmentError("VECTOR_DB_ENDPOINT not configured.")

    # 1. Initialize OpenSearch Connection with IAM Auth
    auth = get_aws_auth_header()
    opensearch_client = OpenSearch(
        hosts=[{'host': VECTOR_DB_ENDPOINT, 'port': 443}],
        http_auth=auth,
        use_ssl=True,
        verify_certs=True,
        connection_class=RequestsHttpConnection
    )
    
    # 2. Initialize LangChain Vector Store
    vectorstore = OpenSearchVectorSearch(
        index_name=OPENSEARCH_INDEX_NAME,
        embedding_function=embeddings_model,
        opensearch_url=f"https://{VECTOR_DB_ENDPOINT}",
        client=opensearch_client
    )

    # 3. Create RAG Chain/Retriever
    llm = Bedrock(client=bedrock_client, model_id=BEDROCK_MODEL_ID)
    
    # Use the vector store as the retriever
    retriever = vectorstore.as_retriever(search_kwargs={"k": 3}) 
    
    return RetrievalQA.from_chain_type(
        llm=llm, 
        chain_type="stuff", 
        retriever=retriever,
        return_source_documents=True # Allows the API to return citations to the frontend
    )

# Attempt to initialize the RAG chain on startup
try:
    qa_chain = initialize_retriever()
    # Ping the client to ensure the connection is live
    if qa_chain.retriever.vectorstore.client.ping():
        print("RAG Chain initialized and OpenSearch client ping successful.")
    else:
        raise ConnectionError("OpenSearch ping failed.")
except Exception as e:
    print(f"CRITICAL ERROR during RAG chain initialization: {e}")
    qa_chain = None # Prevents API from accepting queries if initialization fails
    
# --- API Endpoints ---

@app.route('/health')
def health_check():
    """Health check endpoint for the frontend GUI and ALB Target Group."""
    status = "ok"
    db_status = "unreachable"
    
    # Check RAG Chain initialization status
    if not qa_chain:
        status = "degraded"
    
    # Check live DB status
    try:
        # Use a short timeout for the ping
        if qa_chain and qa_chain.retriever.vectorstore.client.ping(): 
            db_status = "ok"
    except Exception:
        status = "degraded"

    return jsonify({
        "status": status, 
        "service": "rag-api",
        "vector_db_status": db_status
    })

@app.route('/query', methods=['POST'])
def handle_query():
    """Processes a user query against the RAG system."""
    if not qa_chain:
        return jsonify({"error": "RAG system is offline or failed initialization."}, 503)

    query = request.json.get('query')
    if not query:
        return jsonify({"error": "Missing 'query' parameter."}, 400)
    
    try:
        # Execute the RAG chain
        # The 'query' key must match the expected input key for RetrievalQA.from_chain_type
        result = qa_chain.invoke({"query": query}) 
        
        # Extract sources from the returned documents
        source_documents = result.get('source_documents', [])
        sources = [
            doc.metadata.get('source_key', 'Unknown Source') 
            for doc in source_documents
        ]
        
        return jsonify({
            "query": query,
            "response": result['result'],
            "source_documents": sources # Used by the Streamlit frontend to display citations
        })

    except Exception as e:
        print(f"Error processing query: {e}")
        return jsonify({"error": "Internal RAG processing error.", "details": str(e)}, 500)


if __name__ == '__main__':
    # This is run by Gunicorn via entrypoint.sh on internal port 5000
    app.run(host='0.0.0.0', port=5000)
