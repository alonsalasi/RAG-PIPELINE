import os
import streamlit as st
import requests
import json

# --- Configuration ---
# Frontend knows the Backend API is running on the internal port 5000
API_BASE_URL = "http://localhost:5000" 
QUERY_URL = f"{API_BASE_URL}/query"
HEALTH_URL = f"{API_BASE_URL}/health"


def send_query(query: str):
    """Sends the user query to the RAG API endpoint."""
    headers = {"Content-Type": "application/json"}
    payload = {"query": query}
    
    try:
        response = requests.post(QUERY_URL, headers=headers, data=json.dumps(payload))
        response.raise_for_status() # Raise HTTPError for bad responses (4xx or 5xx)
        return response.json()
    except requests.exceptions.RequestException as e:
        # st.error(f"Error connecting to the RAG API: {e}")
        return None

def check_backend_health():
    """Checks the health of the backend API."""
    try:
        response = requests.get(HEALTH_URL, timeout=3)
        if response.status_code == 200:
            return response.json().get('status', 'ok')
        return "Degraded (API Status Error)"
    except requests.exceptions.RequestException:
        return "Down (API Connection Failed)"


# --- Streamlit UI ---

st.set_page_config(page_title="RAG Chatbot", layout="centered")
st.title("🤖 AWS RAG Document Chatbot")

# Display Backend Health Status
health_status = check_backend_health()
st.sidebar.markdown(f"**Backend API Status:** `{health_status}`")

# Initialize chat history
if "messages" not in st.session_state:
    st.session_state["messages"] = [
        {"role": "assistant", "content": "Hello! I can answer questions from your documents. Ask me anything."}
    ]

# Display chat messages from history on app rerun
for message in st.session_state.messages:
    with st.chat_message(message["role"]):
        st.markdown(message["content"])

# Handle user input
if prompt := st.chat_input("Ask a question about your documents..."):
    # 1. Add user message to history and display
    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.markdown(prompt)

    # 2. Get AI response
    with st.chat_message("assistant"):
        with st.spinner("Searching and generating response..."):
            
            # Send query to the API backend
            api_response = send_query(prompt)
            
            if api_response and api_response.get('response'):
                response_text = api_response['response']
                sources = api_response.get('source_documents', [])
                
                # Format output with sources
                full_response = f"{response_text}\n\n"
                if sources:
                    unique_sources = set(sources)
                    full_response += "**Sources:**\n"
                    for source in unique_sources:
                        full_response += f"- `{source}`\n" 
                
                st.markdown(full_response)
                st.session_state.messages.append({"role": "assistant", "content": full_response})
            else:
                error_message = api_response.get('error', 'Could not connect or process query in backend API.')
                st.error(error_message)
                st.session_state.messages.append({"role": "assistant", "content": f"Error: {error_message}"})
